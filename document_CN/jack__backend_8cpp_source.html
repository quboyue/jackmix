61 bool JackBackend::addOutput( QString name ) {
62  if ( client ) {
63  qDebug() << "JackBackend::addOutput(" << name << ")";
64  /*
65  * We need to register inputs by creating entries
66  * in the involumes hash if the interpolation functions
67  * are to be used. Just referencing them is sufficient
68  * to create a partially initialised FaderState.
69  *
70  * setOutput() will finish the job when the output level
71  * is set.
72  */

77  return true;
78  }

139 bool JackBackend::renameOutput(const QString old_name, const QString new_name) {
140  bool done_it = renameOutput(old_name, new_name.toStdString().c_str());
141 
142  if (done_it) {
143  if (outvolumes.contains(old_name)) {
144  outvolumes.insert(new_name, outvolumes[old_name]);
145  outvolumes.remove(old_name);
146  }
147  QStringListIterator ipi(in_ports_list);
148  while(ipi.hasNext()) {
149  QString input(ipi.next());
150  if (volumes[input].contains(old_name)) {
151  volumes[input].insert(new_name, volumes[input][old_name]);
152  volumes[input].remove(old_name);
153  }
154  }
155  }
156 
157  return done_it;

162 bool JackBackend::rename(portsmap &map, QStringList &lst, const QString old_name, const char *new_name) {
163  bool done_it = false;
164 
165  if (client) {
166  ports_cit port_i;
167  // Check to see if the name is in this map
168  if ((port_i=map.constFind(old_name)) != map.constEnd()) {
169  // Found it... try to rename the jack port
170  jack_port_t *port = port_i.value();
171  done_it = (jack_port_rename(client, port, new_name) == 0);
172  // If that works, go on to update our port map
173  if (done_it) {
174  map.insert(new_name, port);
175  map.remove(old_name);
176  // We have to maintain the port lists as well as the maps
177  int pos;
178  if ((pos = lst.indexOf(old_name)) >= 0)
179  lst[pos] = new_name;
180  qDebug()<<"JackBackend::rename -- list now"<<lst;
181  }
182  }
183  }
184 
185  return done_it;
186 }

189 void JackBackend::setVolume(QString channel, QString output, float volume) {
190  qDebug() << "JackBackend::setVolume( " << channel << ", " << output << ", " << volume << " )";
191  if ( channel == output ) {
192  // addInput or addOutput will have created uninitialised FaderState entries,
193  // so we can use that in the following test.
194  if ( involumes.contains( channel ) )
195  setInVolume(channel, volume);
196  else
197  setOutVolume(channel, volume);
198  } else {
199  if (!volumes[channel].contains(output)) {
200  qDebug() << "Creating new FaderState. interp_len = " << interp_len;
201  volumes[channel].insert(output, FaderState(volume, this));
202  } else
203  volumes[channel][output] = volume;
204  }
205 }

272 int JackMix::process( jack_nframes_t nframes, void* arg ) {
273  //qDebug() << "------------------------Jack Process Start for one loop";
274  //qDebug() << "----JackMix::process( jack_nframes_t " << nframes << ", void* )";
275  JackMix::JackBackend* backend = static_cast<JackMix::JackBackend*>( arg );
276 
277  // Deal with MIDI events
278  void* midi_buffer { ::jack_port_get_buffer(backend->midi_port, nframes) };
279  ::jack_midi_event_t event;
280  ::jack_nframes_t event_count { jack_midi_get_event_count(midi_buffer) };
281 
282  for (::jack_nframes_t e {0}; e < event_count; e++) {
283  ::jack_midi_event_get(&event, midi_buffer, e);
284  if (event.buffer[0] == 0xb0) // It's a control change message
285  backend->send_signal(event.buffer[1], event.buffer[2]);
286  }
287 
288  QMap<QString,jack_default_audio_sample_t*> ins;
289  JackMix::ports_it it;
290  for ( it = backend->in_ports.begin(); it!=backend->in_ports.end(); ++it )
291  ins.insert( it.key(), (jack_default_audio_sample_t*) jack_port_get_buffer( it.value(), nframes ) );
292  QMap<QString,jack_default_audio_sample_t*> outs;
293  for ( it = backend->out_ports.begin(); it != backend->out_ports.end(); ++it )
294  outs.insert( it.key(), (jack_default_audio_sample_t*) jack_port_get_buffer( it.value(), nframes ) );
295  ports_it in_it;
296  ports_it out_it;
298  for ( out_it = backend->out_ports.begin(); out_it != backend->out_ports.end(); ++out_it ) {
299  jack_default_audio_sample_t* tmp = outs[ out_it.key() ];
300  for ( jack_nframes_t n=0; n<nframes; n++ ) tmp[ n ] = 0;
301  }
303  for ( in_it = backend->in_ports.begin(); in_it != backend->in_ports.end(); ++in_it ) {
304  QString key {in_it.key()};
305  jack_default_audio_sample_t* tmp = ins[ key ];
306 
307  float max =
308  backend->interp_fader<jack_default_audio_sample_t>(
309  tmp, nframes, backend->getInVolume(key)
310  );
311 
312  backend->newInputLevel(key, max);
313  }
314 
316  for ( in_it = backend->in_ports.begin(); in_it != backend->in_ports.end(); ++in_it ) {
317  jack_default_audio_sample_t* tmpin = ins[ in_it.key() ];
318  for ( out_it = backend->out_ports.begin(); out_it != backend->out_ports.end(); ++out_it ) {
319  jack_default_audio_sample_t* tmpout = outs[ out_it.key() ];
320  backend->interp_fader<jack_default_audio_sample_t>(
321  tmpout, tmpin, nframes, backend->getMatrixVolume(in_it.key(), out_it.key())
322  );
323  }
324 
325  }
327  for ( out_it = backend->out_ports.begin(); out_it != backend->out_ports.end(); ++out_it ) {
328  QString key {out_it.key()};
329  jack_default_audio_sample_t* tmp = outs[ key ];
330 
331  float max =
332  backend->interp_fader<jack_default_audio_sample_t>(
333  tmp, nframes, backend->getOutVolume(key)
334  );
335 
336  backend->newOutputLevel(key, max);
337  }
338 
339  // Send any information about channel levels
340  backend->report();
341  qDebug() << "------------------------Jack Process end for one loop";
342  return 0;
343 }
