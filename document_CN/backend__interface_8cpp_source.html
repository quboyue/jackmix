40 BackendInterface::Level BackendInterface::signalToLevel(float sig) const
41 {
42  if (sig >= threshold[too_high]) return too_high;

48 void BackendInterface::newLevel(Stats& s, float maxSignal)
49 {
50  Level levelNow = signalToLevel(maxSignal);
51  if (levelNow == too_high) s.timeout.start();
52  if (s.level == too_high && s.timeout.elapsed() < 1000)
53  s.changed = (s.level != too_high);
54  else {
55  s.changed = (s.level != levelNow);
56  s.level = levelNow;
57  }
58 }

75 void BackendInterface::report_group(int which, levels_t& result)
76 {
77  QMap<QString, Stats>::const_iterator i = stats[which].constBegin();
78  while (i != stats[which].constEnd()) {
79  const Stats* v = &i.value();
80  if (v->changed) {
81  result[i.key()] = v->level;
82  }
83  ++i;
84  }

117 BackendInterface::FaderState& BackendInterface::FaderState::operator=(float volume) {
118 
119  if (!qFuzzyCompare(current, target)) {
120  // Time to change current in case we're not finished interpolating
