44 void AbstractSlider::value( double v ) {
45  if ( !_value_inupdate ) {
46  v = qMin( v, dbmax );
47  v = qMax( v, dbmin );
48  _value = v;
49  update();
50  _value_inupdate = true;
51  emit valueChanged( _value );
52  _value_inupdate = false;
53  }
54 }

56 void AbstractSlider::setNormalisedValue(double v, bool show_numeric) {
57  v = qMax(v, 0.0);
58  v = qMin(v, 1.0);
59  // The call below will be handled by a virtual method in a derived class
60  // The second argument enables or suppresses an auxilliary numeric display
61  // of the slider's value.
62  value(dbmin + v*(dbmax-dbmin), show_numeric);
63 }

65 void AbstractSlider::setMidiValue(int iv) {
66  iv = qMax(iv, 0);
67  iv = qMin(iv, 127);
68  // Some cheap MIDI controllers "dither" (change randomly by 1 or more LSBs)
69  // so we'll suppress any auxilliary display of slider value because it might
70  // keep popping up annoyingly when the slider isn't being adjusted.
71  setNormalisedValue(iv/127.0, false);
72 }

80 void AbstractSlider::showInput() {
81  if ( _spinbox.isNull() ) {
82  _spinbox = new QDoubleSpinBox( this );
83  _spinbox->setMinimum( dbmin );
84  _spinbox->setMaximum( dbmax );
85  _spinbox->setSingleStep( _pagestep );
86  _spinbox->setValue( _value );
87  _spinbox->setFrame( false );
88  connect( _spinbox, SIGNAL( editingFinished() ), this, SLOT( hideInput() ) );
89  //qDebug() << iv << "=>" << ((static_cast<double>(iv))/127.0);
90 
91  connect( _spinbox, SIGNAL( valueChanged( double ) ), this, SLOT( value( double ) ) );
92  connect( this, SIGNAL( valueChanged( double ) ), _spinbox, SLOT( setValue( double ) ) );
93  _spinbox->show();
94  _spinbox->move( ( width()-_spinbox->width() )/2, ( height()-_spinbox->height() )/2 );
95  _spinbox->setFocus();
96  }
97 }

108 void AbstractSlider::mousePressEvent( QMouseEvent* ev ) {
109  qDebug() << "AbstractSlider::mousePressEvent(" << ev << ") is accepted?" << ev->isAccepted();
110  if ( ev->button() == Qt::LeftButton) {
111  qDebug() << "AbstractSlider::mousePressEvent::ev->button()(" << ev->button();
112 
113  if (ev->modifiers() == Qt::ShiftModifier)
114  {
115  qDebug() << "Qt::ShiftModifier ";
116  emit(select());
117  ev->accept();
118  }
119  else if ( ev->modifiers() == (Qt::ShiftModifier | Qt::ControlModifier) )
120  {
121  qDebug() << "Qt::ShiftModifier | Qt::ControlModifier ";
122  emit(replace());
123  ev->accept();
124  }
125  else
126  {
127  qDebug() << "_nullclick.contains( ev->pos() ) ";
128  if ( _nullclick.contains( ev->pos() ) )
129  value( 0 );
130  else
131  mouseEvent( ev );
132  }
133  }
