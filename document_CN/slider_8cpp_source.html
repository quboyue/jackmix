71 void Slider::paintEvent( QPaintEvent* ) {
72  bool rotated = false;
73  QStyleOption opt;
74  QPainter p( this );
75  p.setRenderHints( QPainter::Antialiasing );
76  QString tmp = QString::number( _value );
77  if ( tmp.contains( "." ) ) tmp = _valuestring.arg( tmp.left( tmp.indexOf( "." ) + _precision + 1 ) );
78  else tmp = _valuestring.arg( tmp );
79 
80  QFontMetrics metrics( font() );
81  int fontwidth = metrics.width( tmp );
82 
83  // Center the coordinates
84  p.translate( width()/2, height()/2 );
85 
86  double w = width()-SLIDER_BORDER;
87  double h = height()-SLIDER_BORDER;
88  if ( width() < height() ) {
89  w = height()-SLIDER_BORDER;
90  h = width()-SLIDER_BORDER;
91  p.rotate( -90 );
92  rotated = true;
93  }
94 
95  if ( hasFocus() ) {
96  style()->drawPrimitive( QStyle::PE_FrameFocusRect, &opt, &p, this );
97  }
98 
99  //double pos = dbtondb( _value )*w-w/2;
100  QRectF bar( -w/2, -h/4, w, h/2 );
101 
102  // marks
103  p.save();
104  p.setPen( palette().color( QPalette::ButtonText ) );
105  QFont small = font();
106  small.setPointSizeF( qMax( qreal(5.0), font().pointSizeF()/2.0 ) );
107  p.setFont( small );
108  if ( _show_value ) {
109  for ( double a=_pagestep; a<dbmax; a+=_pagestep )
110  p.drawLine( QPointF( -w/2+w*dbtondb( a ), h/3.5 ), QPointF( -w/2+w*dbtondb( a ), -h/3.5 ) );
111  for ( double a=-_pagestep; a>dbmin; a-=_pagestep )
112  p.drawLine( QPointF( -w/2+w*dbtondb( a ), h/3.5 ), QPointF( -w/2+w*dbtondb( a ), -h/3.5 ) );
113  }
114  QList<double> _texts;
115  _texts << dbmin << dbmax;
116  if ( 0>dbmin && 0<dbmax )
117  _texts << 0.0;
118  foreach( double a, _texts ) {
119  p.save();
120  p.translate( -w/2+w*dbtondb( a ), 0 );
121  p.drawLine( QPointF( 0, h/3 ), QPointF( 0, -h/3 ) );
122  QRectF rect(
123  0,0,
124  QFontMetrics( small ).width( _valuestring ),
125  QFontMetrics( small ).height() );
126  if ( !rotated ) {
127  if ( dbtondb( a ) > 0.5 )
128  rect.translate( -QFontMetrics( small ).width( _valuestring ), 0 );
129  p.drawText( rect.translated( 0, h/3 ), Qt::AlignCenter, QString( _valuestring ).arg( a ) );
130  //p.drawText( rect.translated( 0, -h/3 -rect.height() ), Qt::AlignCenter, QString( _valuestring ).arg( a ) );
131  if ( a == 0.0 ) {
132  _nullclick = p.matrix().mapRect( rect.translated( 0, h/3 ) ).toRect();
133  //_nullclick = _nullclick.united( p.matrix().mapRect( rect.translated( 0, -h/3 -rect.height() ) ).toRect() );
134  }
135  } else {
136  p.rotate( 90 );
137  if ( dbtondb( a ) < 0.5 )
138  rect.translate( 0, -QFontMetrics( small ).height() );
139  p.drawText( rect.translated( h/3, 0 ), Qt::AlignCenter, QString( _valuestring ).arg( a ) );
140  //p.drawText( rect.translated( -h/3 -rect.width(), 0 ), Qt::AlignCenter, QString( _valuestring ).arg( a ) );
141  if ( a == 0.0 ) {
142  _nullclick = p.matrix().mapRect( rect.translated( h/3,0 ) ).toRect();
143  //_nullclick = _nullclick.united( p.matrix().mapRect( rect.translated( -h/3-rect.width(),0 ) ).toRect() );
144  }
145  }
146  p.restore();
147  }
148  p.restore();
149 
150  // Surrounding rect
151  p.drawRect( bar );
152 
153  // Rect for the whole bar
154  {
155  p.save();
156  QLinearGradient grad( QPointF( -w/2, -h/4 ), QPointF( w/2, -h/4 ) );
157  // Global ends first
158  grad.setColorAt( 0.0, palette().color( QPalette::Highlight ) );
159  if ( dbtondb( _value ) < 1.0 )
160  grad.setColorAt( 1.0, palette().color( QPalette::Highlight ).dark() );
161  // Next soft-fades
162  grad.setColorAt( qMax( 0.0, dbtondb( _value )-0.01 ), palette().color( QPalette::Highlight ).light() );
163  if ( dbtondb( _value )+0.01 < 1.0 )
164  grad.setColorAt( qMin( 1.0, dbtondb( _value )+0.01 ), palette().color( QPalette::Highlight ) );
165  // Last the value itself
166  grad.setColorAt( dbtondb( _value ), palette().color( QPalette::HighlightedText ) );
167  // That way minimum and maximum get the right color
168  p.fillRect( bar, grad );
169  p.restore();
170  }
171 
172  // Set _faderarea correctly
173  _faderarea = p.matrix().mapRect( bar ).toRect();
174 
175  // de-rotate
176  if ( rotated )
177  p.rotate( 90 );
178 
179  // Draw the value as text in the lower third
180  if ( _show_value ) {
181  p.save();
182  p.setPen( Qt::NoPen );
183  p.setBrush( palette().color( QPalette::Base ) );
184  p.setOpacity( 0.75 );
185  p.drawRoundRect( -fontwidth/2 -2, height()/3 -metrics.ascent() -1, fontwidth +4, metrics.ascent() +4 );
186  p.restore();
187  // Text showing the value
188  p.drawText( -fontwidth/2, height()/3, tmp );
189  }
190 }
