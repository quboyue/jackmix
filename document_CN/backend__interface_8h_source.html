110  // The BackendInterface interface is assumed present in all backends.
111  // This is responsible for making the flashy lights on the input
112  // and output pots change colour. If you don't want to implement
113  // that you can ignore it, and no signals will be sent; the input
114  // and output potentiometers remain dark
115  //
116  // If you would like to impelment it, the idea is to record the
117  // abs maximum level of each buffer (as floats in the range 0..1)
118  // newInputLevel(QString key, float max) and newOutputLevel will
119  // do that for you. When the levels for each input and output have
120  // been recorded, call report() to signal the Widget to change the
121  // colours of the input and output elements
122  //
123  // Similarly, of the backend of the dervied class uses its own
124  // MIDI message handler, it can use the cc_message signal to notify
125  // control changes.

249  template <typename SampT>
250  inline SampT interp_fader(SampT* buf, const size_t nframes, FaderState& fs) {
251  SampT max {0};
252 
253  if ( fs.p != nullptr && !qFuzzyCompare(fs.target, fs.current)) {
254 // qDebug() << "In/Out fader. Target " << fs.target
255 // << ": " << fs.num_steps << " steps, now " << fs.cur_step ;
256 
257  for (size_t n {0}; n < nframes; n++) {
258  buf[n] *= fs.current + fs.cur_step*(fs.target - fs.current)/fs.num_steps;
259  max = qMax(max, buf[n]);
260  fs.cur_step += (fs.cur_step < fs.num_steps);
261  }
262 
263  if (fs.cur_step == fs.num_steps) {
264  qDebug() << "In/Out fader reached target at step " << fs.cur_step ;
265  fs.current = fs.target;
266  fs.num_steps = interp_len; // in case sample rate has changed
267  fs.cur_step = 0; // for safety
268  }
269 
270  } else {
271  //qDebug() << "No interp. current = " << fs.current ;
272  for (size_t n {0}; n < nframes; n++) {
273  buf[n] *= fs.current;
274  max = qMax(max, buf[n]);
275  }
276  }
277  return max;
278  }
