42 void PortListener::run()
43 {
44  int npfd = snd_seq_poll_descriptors_count(seq_handle, POLLIN);
45 
46  // Using C style memory allocation for communication with the kernel-level MIDI system
47  struct pollfd *pfd = (struct pollfd *)alloca(npfd * sizeof(struct pollfd));
48  snd_seq_poll_descriptors(seq_handle, pfd, npfd, POLLIN);
49  snd_seq_event_t *ev;
50  int param;
51  int value;
52 
53  running = true;
54 
55  while (running) {
56  // *ix system calls are uninterruptable, so we'll pop back into
57  // user-space every 200ms or so
58  if (poll(pfd, npfd, 200) > 0) {
59  do {
60  snd_seq_event_input(seq_handle, &ev);
61  if (ev -> type == SND_SEQ_EVENT_CONTROLLER) {
62  param = ev->data.control.param;
63  value = ev->data.control.value;
64  emit message(param, value);
65  }
66  snd_seq_free_event(ev);
67  } while (snd_seq_event_input_pending(seq_handle, 0) > 0);
68  }
69 
70  qDebug() << "Tick";
71 
72  }
73 
74  exit(0);
75 }

77 ControlSender::ControlSender(const char* port_name)
78  : have_alsa_seq {false}
79  , port_listener {nullptr}
80 {
81 
82  if ((have_alsa_seq = snd_seq_open(&seq_handle, "default", SND_SEQ_OPEN_DUPLEX, 0)) == 0) {
83  // Name the client
84  snd_seq_set_client_name(seq_handle, port_name);
85 
86  // Open one input port
87  if ((port_id = snd_seq_create_simple_port(seq_handle, "Control Input",
88  SND_SEQ_PORT_CAP_WRITE | SND_SEQ_PORT_CAP_SUBS_WRITE,
89  SND_SEQ_PORT_TYPE_APPLICATION)) < 0)
90  throw new MidiControlException("Can't open MIDI port");
91 
92  port_listener = new JackMix::MidiControl::PortListener(seq_handle, port_id);
93  port_listener->start();
94  connect (port_listener, SIGNAL(message(int, int)), this, SLOT(despatch_message(int, int)) );
95  } else
96  seq_handle = nullptr;
97 }

99 ControlSender::~ControlSender() {
100  // close down the port listner thread
101  port_listener->quit();
102  port_listener->wait();
103  // then close the MIDI port
104  if (have_alsa_seq) {
105  snd_seq_delete_simple_port(seq_handle, port_id);
106  snd_seq_close(seq_handle);
107  }
108 }
