47  int levels = n.count( "/" ); // + 1
48  if ( n.section( "/",0,0 ) == _name ) {
49  // The first part is our name, go on...
50  if ( levels == 0 )
51  return this;
52  else {
53  QString tmp = n.section( "/", 1,levels );
54  ConfigOption* ret = 0;
55  for ( ConfigOption* it=_childs.first(); it && ret == 0; it++ ) {
56  ret = it->getOption( tmp );
57  }
58  return ret;
59  }
60  }
61  else // first part isn't our name, return 0
62  return 0;
