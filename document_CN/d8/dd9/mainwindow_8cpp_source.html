236 void MainWindow::openFile( QString path ) {
237  qDebug() << "MainWindow::openFile(" << path << ")";
238  if ( path.isEmpty() )
239  return;
240 
241  // delay autofill at least until all saved elements are created:
242  bool save_autofillscheduled = _autofillscheduled;
243  _autofillscheduled = true;
244 
245  QFile file( path );
246  if ( file.open( QIODevice::ReadOnly | QIODevice::Text ) ) {
247  while ( _backend->outchannels().size() > 0 )
248  removeOutput( _backend->outchannels()[ 0 ] );
249  while ( _backend->inchannels().size() > 0 )
250  removeInput( _backend->inchannels()[ 0 ] );
251 
252  QDomDocument doc( "bla" );
253  doc.setContent( &file );
254 
255  QDomElement jackmix = doc.documentElement();
256  QString version = jackmix.attribute( "version", JACKMIX_FILE_FORMAT_VERSION );
257 
258  if ( version == JACKMIX_FILE_FORMAT_VERSION ) {
259 
260  QDomElement ins = jackmix.firstChildElement( "ins" );
261  for ( QDomElement in = ins.firstChildElement( "channel" ); !in.isNull(); in = in.nextSiblingElement( "channel" ) ) {
262  QString name = in.attribute( "name" );
263  QString volume = in.attribute( "volume" );
264  QString midi = in.attribute( "midi" );
265  addInput( name );
266  _backend->setVolume( name, name, volume.toDouble() );
267  // Input gain elements have the same name for input and output
268  _inputmps[QString("%1,%1").arg(name)] = midi;
269 
270  }
271 
272  QDomElement outs = jackmix.firstChildElement( "outs" );
273  for ( QDomElement out = outs.firstChildElement( "channel" ); !out.isNull(); out = out.nextSiblingElement( "channel" ) ) {
274  QString name = out.attribute( "name" );
275  QString volume = out.attribute( "volume" );
276  QString midi = out.attribute("midi");
277  addOutput( name );
278  _backend->setVolume( name, name, volume.toDouble() );
279  // Output gain elements have the same name for input and output
280  _outputmps[QString("%1,%1").arg(name)] = midi;
281  }
282 
283  QDomElement matrix = jackmix.firstChildElement( "matrix" );
284  for ( QDomElement volume = matrix.firstChildElement( "volume" ); !volume.isNull(); volume = volume.nextSiblingElement( "volume" ) ) {
285  QString in = volume.attribute( "in" );
286  QString out = volume.attribute( "out" );
287  QString value = volume.attribute( "value" );
288  QString midi = volume.attribute( "midi" );
289  _backend->setVolume( in, out, value.toDouble() );
290  if (!midi.isEmpty()) _mixermps[QString("%1,%2").arg(in, out)] = midi;
291  }
292 
293  QDomElement elements = jackmix.firstChildElement( "elements" );
294  for ( QDomElement element = elements.firstChildElement( "element" ); !element.isNull(); element = element.nextSiblingElement( "element" ) ) {
295  QStringList ins;
296  for ( QDomElement in = element.firstChildElement( "in" ); !in.isNull(); in = in.nextSiblingElement( "in" ) )
297  ins << in.attribute( "name" );
298  QStringList outs;
299  for ( QDomElement out = element.firstChildElement( "out" ); !out.isNull(); out = out.nextSiblingElement( "out" ) )
300  outs << out.attribute( "name" );
301  _mixerwidget->createControl( ins, outs );
302  // Just one set of midi control parameters for the whole saved element
303  _mixermps[QString("%1,%2").arg(ins[0], outs[0])] = element.attribute("midi");
304 
305  }
306  }
307 
308  file.close();
309  }
310 
311  _autofillscheduled = save_autofillscheduled;
312  scheduleAutoFill();
313  qDebug() << "MainWindow::openFile() finished";
314 }

316 void MainWindow::updateAutoFilledMidiParams(MixingMatrix::Widget *w) {
317  QHash<QString,QString> *mphash(0); // Iterate over the right midi parameter set
318  qDebug() << "Autofill is complete. for widget" << w;
319  if (w == _inputswidget) {
320  qDebug("(inputs widget)");
321  mphash = &_inputmps;
322  } else if (w == _outputswidget) {
323  qDebug("(outputs widget)");
324  mphash = &_outputmps;
325  } else if (w == _mixerwidget) {
326  qDebug("(mixer widget)");
327  mphash = &_mixermps;
328  } else { qDebug("(UNKNOWN widget)");
329  }
330 
331  qDebug("Setting %d MIDI control parameters.", mphash->size());
332  QMutableHashIterator<QString,QString> mpiter(*mphash);
333  while (mpiter.hasNext()) {
334  mpiter.next();
335  QString name = mpiter.key();
336  const QStringList posn(name.split(","));
337  qDebug() << "posn=" << posn;
338  Element *el(w->getResponsible(posn[0], posn[1]));
339  if (el) {
340  const QStringList params( (mpiter.value()).split(",") );
341  QList<int> pv;
342  qDebug()<<name<<" has parameters "<<params;
343  for (int p = 0; p < params.size(); p++)
344  pv.append(params[p].toInt());
345  el->update_midi_parameters(pv);
346  // We're going to remove elements from the hash one at a time because
347  // there might be some awaiting initialisation. If we clear them all
348  // at the end, we'll stand a good chance of never setting the MIDI
349  // parameters on some of them.
350  mpiter.remove();
351  } else {
352  // This behaviour is actually quite normal; it happens for example after the saved
353  // compound elements are regenerated and autoFill'd, but before the simple
354  // aux_elements have got going.
355  qDebug() << name
356  << "has null responsible element but has assigned midi parameters "
357  << mpiter.value();
358  }
359  }

362 void MainWindow::saveFile( QString path ) {
363  if ( path.isEmpty() )
364  path = QFileDialog::getSaveFileName( this, 0, 0, "JackMix-XML (*.jm-xml)" );
365  if ( path.isEmpty() )
366  return;
367 
368  if ( ! path.endsWith( ".jm-xml" ) )
369  path += ".jm-xml";
370 
371  QStringList ins = _backend->inchannels();
372  QStringList outs = _backend->outchannels();
373 
374  qDebug() << "Saving file. Using ins/outs lists: " << ins << outs;
375  QString xml = "<jackmix version=\"" JACKMIX_FILE_FORMAT_VERSION "\"><ins>";
376  foreach( QString in, ins ) {
377  xml += QString( "<channel name=\"%1\" volume=\"%2\" midi=\"" )
378  .arg( in ).arg( _backend->getVolume( in,in ) );
379  qDebug()<<" Responsible element for " << in << ": " << _inputswidget->getResponsible(in, in);
380  const QList<int> &mp = _inputswidget->getResponsible(in,in)->midiParameters();
381  // I'm going to use a loop to make it clear ordering is important
382  // (actually, foreach maitains ordering of a QList in Qt 4.8, but this might change
383  QStringList mpl;
384  for (int i = 0 ; i < mp.size() ; i++ )
385  mpl << QString("%1").arg(mp.at(i));
386  xml += mpl.join(",") + "\" />";
387  }
388  xml += "</ins><outs>";
389  foreach( QString out, outs ) {
390  xml += QString( "<channel name=\"%1\" volume=\"%2\" midi=\"" )
391  .arg( out ).arg( _backend->getVolume( out,out ) );
392  const QList<int> &mp = _outputswidget->getResponsible(out,out)->midiParameters();
393  QStringList mpl;
394  for (int i = 0 ; i < mp.size() ; i++ )
395  mpl << QString("%1").arg(mp.at(i));
396  xml += mpl.join(",") + "\" />";
397  }
398  xml += "</outs><matrix>";
399  foreach( QString in, ins )
400  foreach( QString out, outs ) {
401  xml += QString( "<volume in=\"%1\" out=\"%2\" value=\"%3\"" )
402  .arg( in ).arg( out ).arg( _backend->getVolume( in, out ) );
403  const Element *el = _mixerwidget->getResponsible(in,out);
404  // Don't bother writing out midi parameter numbers for compound elements here
405  if (el->in().size() == 1 && el->out().size() == 1) {
406  const QList<int> &mp = el->midiParameters();
407  QStringList mpl;
408  for (int i = 0 ; i < mp.size() ; i++ )
409  mpl << QString("%1").arg(mp.at(i));
410  xml += " midi=\"" + mpl.join(",") + "\"";
411  }
412  xml += " />";
413  }
414  xml += "</matrix>";
415 
416  xml += "<elements>";
417  // Elements
418  QList<Element*> savedelements;
419  foreach( QString in, ins )
420  foreach( QString out, outs ) {
421  Element* tmp = _mixerwidget->getResponsible( in, out );
422  if ( !savedelements.contains( tmp ) && ( tmp->in().size() > 1 || tmp->out().size() > 1 ) ) {
423  // Save all the midi conotrol parameters for this compound element
424  xml += "<element midi=\"";
425  const QList<int> &mp = tmp->midiParameters();
426  QStringList mpl;
427  for (int i = 0 ; i < mp.size() ; i++ )
428  mpl << QString("%1").arg(mp.at(i));
429  xml += mpl.join(",") + "\">";
430  foreach( QString input, tmp->in() )
431  xml += QString( "<in name=\"%1\" />" ).arg( input );
432  foreach( QString output, tmp->out() )
433  xml += QString( "<out name=\"%1\" />" ).arg( output );
434  xml += "</element>";
435  savedelements.push_back( tmp );
436  }
437  }
438  xml += "</elements>";
439 
440  xml += "</jackmix>";
441 
442  QFile file( path );
443  if ( file.open( QIODevice::WriteOnly | QIODevice::Text ) ) {
444  QTextStream stream( &file );
445  stream << xml.replace( ">", ">\n" );
446  file.close();
447  }

547 void MainWindow::renameInput( QStringList names ) {
548  qDebug() << "Change input channel names: " << names;
549  while (names.length() > 0) {
550  QString old_name(names.takeFirst());
