74 Mono2StereoElement::Mono2StereoElement( QStringList inchannel, QStringList outchannels, MixingMatrix::Widget* p, const char* n )
75  : Element( inchannel, outchannels, p, n )
76  , dB2VolCalc( -42, 6 )
77  , _balance_value( 0 )
78  , _volume_value( 0 )
79 {
80  qDebug( "Mono2StereoElement::Mono2StereoElement()" );
81  double left = backend()->getVolume( _in[0], _out[0] );
82  double right = backend()->getVolume( _in[0], _out[1] );
83  qDebug( " volumes: %f, %f", left, right );
84  if ( left>right ) {
85  _volume_value = left;
86  _balance_value = right-left;
87  } else {
88  _volume_value = right;
89  _balance_value = right-left;
90  }
91  qDebug( " values: %f, %f", _volume_value, _balance_value );
92  QVBoxLayout* _layout = new QVBoxLayout( this );
93  _layout->setMargin( 0 );
94  _layout->setSpacing( 0 );
95 
96  menu()->addAction( "Select", this, SLOT( slot_simple_select() ) );
97  menu()->addAction( "Replace", this, SLOT( slot_simple_replace() ) );
98  menu()->addAction( "Explode", this, SLOT( slot_simple_explode() ) );
99  menu()->addAction( "Assign MIDI Parameter", this, SLOT( slot_assign_midi_parameters() ) );
100 
101  _balance = new JackMix::GUI::Knob( _balance_value, -1, 1, 2, 0.1, this, "%1" );
102  _layout->addWidget( _balance, 10 );
103  connect( _balance, SIGNAL( valueChanged( double ) ), this, SLOT( balance( double ) ) );
104  connect( _balance, SIGNAL( select() ), this, SLOT( slot_simple_select() ) );
105  connect( _balance, SIGNAL( replace() ), this, SLOT( slot_simple_replace() ) );
106  _volume = new JackMix::GUI::Slider( amptodb( _volume_value ), dbmin, dbmax, 2, 3, this );
107  _layout->addWidget( _volume, 20 );
108  connect( _volume, SIGNAL( valueChanged( double ) ), this, SLOT( volume( double ) ) );
109  connect( _volume, SIGNAL( select() ), this, SLOT( slot_simple_select() ) );
110  connect( _volume, SIGNAL( replace() ), this, SLOT( slot_simple_replace() ) );
111 
112  // WATCH OUT: Order of initialisation is really important!
113  // Make sure all the widgets are contructed before adding them to the delegates list
114 
115  // Initial MIDI parameters and associated AbstractSliders
116  midi_params.append(0);
117  midi_delegates.append(_volume);
118  midi_params.append(0);
119  midi_delegates.append(_balance);
120 
121  // Now construct the parameter setting menu
122  _cca = new JackMix::GUI::MidiControlChannelAssigner(QString("Set MIDI control parameter"),
123  "<qt>" + _in[0] + " &rarr; (" + _out[0] + "/" + _out[1] + ")</qt>",
124  QStringList() << "Gain" << "Pan",
125  midi_params,
126  this
127  );
128  connect( _cca, SIGNAL(assignParameters(QList<int>)), this, SLOT(update_midi_parameters(QList<int>)) );
129 
130 }

149 void Mono2StereoElement::calculateVolumes() {
150  double left, right;
151  left = dbtoamp( _volume_value );
152  right = dbtoamp( _volume_value );
153  if ( _balance_value > 0 )
154  left = dbtoamp( _volume_value )*( 1-_balance_value );
155  if ( _balance_value < 0 )
156  right = dbtoamp( _volume_value )*( 1+_balance_value );
157  backend()->setVolume( _in[0], _out[0], left );
158  backend()->setVolume( _in[0], _out[1], right );
159 }

164 Stereo2StereoElement::Stereo2StereoElement( QStringList inchannels, QStringList outchannels, MixingMatrix::Widget* p, const char* n )
165  : Element( inchannels, outchannels, p, n )
166  , dB2VolCalc( -42, 6 )
167  , _balance_value( 0 )
168  , _volume_value( 0 )
169 {
170  backend()->setVolume( _in[0], _out[1], 0 );
171  backend()->setVolume( _in[1], _out[0], 0 );
172  double left = backend()->getVolume( _in[0], _out[0] );
173  double right = backend()->getVolume( _in[1], _out[1] );
174  if ( left>right )
175  _volume_value = left;
176  else
177  _volume_value = right;
178  _balance_value = qMin( qMax( right-left, -1.0 ), 1.0 );
179  QGridLayout* _layout = new QGridLayout( this );
180  _layout->setSpacing( 1 );
181  _layout->setMargin( 2 );
182  _balance_widget = new JackMix::GUI::Slider( _balance_value, -1, 1, 2, 0.1, this, "%1" );
183  _layout->addWidget( _balance_widget, 0,0 );
184  _layout->setRowStretch( 0, 0 );
185  connect( _balance_widget, SIGNAL( valueChanged( double ) ), this, SLOT( balance( double ) ) );
186  connect( _balance_widget, SIGNAL( select() ), this, SLOT( slot_simple_select() ) );
187  connect( _balance_widget, SIGNAL( replace() ), this, SLOT( slot_simple_replace() ) );
188  _volume_widget = new JackMix::GUI::Slider( amptodb( _volume_value ), dbmin, dbmax, 1, 3, this );
189  _layout->addWidget( _volume_widget, 1,0 );
190  _layout->setRowStretch( 1, 255 );
191  connect( _volume_widget, SIGNAL( valueChanged( double ) ), this, SLOT( volume( double ) ) );
192  connect( _volume_widget, SIGNAL( select() ), this, SLOT( slot_simple_select() ) );
193  connect( _volume_widget, SIGNAL( replace() ), this, SLOT( slot_simple_replace() ) );
