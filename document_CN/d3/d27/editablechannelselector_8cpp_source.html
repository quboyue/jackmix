32 EditableChannelSelector::EditableChannelSelector( QString title
33  , QString label
34  , QStringList channels
35  , QWidget *p
36  , const char* n )
37  : ChannelSelector(title, label, channels, p, n)
38  , _channels(channels) {
39  // Just like a ChannelSelector, but the user can edit the channel names.
40  _list->setEditTriggers(QAbstractItemView::AllEditTriggers);
41  _items = channels.length();
42  _isEdited = new bool[_items];
43  for (int i = 0; i < _items; i++) {
44  _list->item(i)->setFlags(_list->item(i)->flags() | Qt::ItemIsEditable);
45  _isEdited[i] = false;
46  }
47  connect(_list, SIGNAL(itemChanged(QListWidgetItem*)),
48  this, SLOT(register_changed_item(QListWidgetItem*)) );
49 }

57 void EditableChannelSelector::update_channel_name(int which, QString revised) {
58  qDebug() << "EditableChannelSelector::update_channel_name(" << which
59  << ", " << revised << "): _channels =" << _channels;
60  _channels[which] = revised;
61  _list->item(which)->setText(revised);
62  _isEdited[which] = false;

65 void EditableChannelSelector::register_changed_item(QListWidgetItem *wi) {
66  qDebug() << "Item changed: " << _list->row(wi);
67  int which = _list->row(wi);
68  _isEdited[which] = true;
69 }

71 void EditableChannelSelector::commit() {
72  qDebug("EditableChannelSelector");
73  QStringList changed;
74  for (int c = 0; c < _items; c++)
75  if (_isEdited[c]) {
76  changed.append(_channels[c]);
77  changed.append(_channels[c] = _list->item(c)->text());
78  _isEdited[c] = false;
79  }
80 
81  emit editedChannels(changed);
82 }
