34  , dbmax( _dbmax )
35  , dbmin( _dbmin )

38  double dbmax, dbmin;
51  double amptodb( double p ) {
52  double db = _base*log10( p );
53  if ( db < dbmin ) db = dbmin;
54  if ( db > dbmax ) db = dbmax;
55  return db;
56  }
57  double dbtoamp( double db ) {
58  double amp = pow( 10, db/_base );
59  if ( db <= dbmin ) amp = 0;
60  return amp;
61  }
63  double amptondb( double p ) {
64  return dbtondb( amptodb( p ) ); //- dbmin ) / ( dbmax - dbmin );
65  }
66  double ndbtoamp( double ndb ) {
67  return dbtoamp( ndb * ( dbmax - dbmin ) + dbmin );
68  }
70  double dbtondb( double db ) {
71  return ( db - dbmin )/( dbmax - dbmin );
72  }
74  double dbtovol( double db ) {
75  return ( db -dbmin )/( 0-dbmin );
76  }
78  double ndbtodb( double ndb ) {
79  return ( ndb * ( dbmax-dbmin ) +dbmin );
80  }
