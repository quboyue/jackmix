94 void qLashClient::timerEvent( QTimerEvent* ) {
95  while ( isConnected() && lash_get_pending_event_count( _client ) > 0 ) {
96  lash_event_t* event = lash_get_event( _client );
97  qDebug() << "Have event of type" << lash_event_get_type( event );
98 
99  switch ( lash_event_get_type( event ) ) {
100  case LASH_Client_Name:
101  qDebug() << "Event: This clients name is:" << lash_event_get_string( event );
102  break;
103  case LASH_Jack_Client_Name:
104  qDebug() << "Event: This clients JACK-name is:" << lash_event_get_string( event );
105  break;
106  case LASH_Alsa_Client_ID:
107  qDebug() << "Event: This clients Alsa-ID is:" << lash_event_get_string( event );
108  break;
109  case LASH_Save_File:
110  qDebug() << "Event: Should save data into dir" << lash_event_get_string( event );
111  emit saveToDir( lash_event_get_string( event ) );
112  emit saveValues();
113  saveToDirFinalize( lash_event_get_string( event ) );
114  break;
115  case LASH_Restore_File:
116  qDebug() << "Event: Should restore data from dir" << lash_event_get_string( event );
117  emit restoreFromDir( lash_event_get_string( event ) );
118  restoreFromDirFinalize( lash_event_get_string( event ) );
119  emit restoreValues();
120  break;
121  case LASH_Save_Data_Set:
122  qDebug() << "Event: Should save data in configs";
123  emit saveToConfig();
124  //emit saveValues(); // The values are saved in their own file in the dir...
125  saveToConfigFinalize();
126  break;
127  case LASH_Restore_Data_Set:
128  qDebug() << "Event: Should restore data from configs";
129  emit restoreFromConfig();
130  restoreFromConfigFinalize();
131  break;
132  case LASH_Quit:
133  qDebug() << "Event: Should terminate now";
134  emit quitApp();
135  break;
136  default:
137  qDebug() << "Handling of event" << lash_event_get_type( event ) << "isn't implemented yet!";
138  }
139 
140  lash_event_destroy( event );
141  }
142 }
