76 AuxElement::AuxElement( QStringList inchannel, QStringList outchannel, MixingMatrix::Widget* p, const char* n )
77  : Element( inchannel, outchannel, p, n )
78  , dB2VolCalc( -42, 6 )
79 {
80  if (p->mode() == Widget::Select) {
81  menu()->addAction( "Select", this, SLOT( slot_simple_select() ) );
82  menu()->addAction( "Replace", this, SLOT( slot_simple_replace() ) );
83  }
84  menu()->addAction( "Assign MIDI Parameter", this, SLOT( slot_assign_midi_parameters() ) );
85 
86  QVBoxLayout* _layout = new QVBoxLayout( this );
87 
88  if ( _in[0] == _out[0] ) {
89  disp_name = new QLabel( QString( "<qt><center>%1</center></qt>" ).arg( _in[0] ), this );
90  _layout->addWidget(disp_name, 0);
91  }
92 
93  _poti = new JackMix::GUI::Knob(
94  amptodb( backend()->getVolume( _in[0], _out[0] ) ),
95  dbmin, dbmax, 2, 3, this );
96  _layout->addWidget( _poti, 100 );
97 
98  connect( _poti, SIGNAL( valueChanged( double ) ), this, SLOT( emitvalue( double ) ) );
99  connect( _poti, SIGNAL( select() ), this, SLOT( slot_simple_select() ) );
100  connect( _poti, SIGNAL( replace() ), this, SLOT( slot_simple_replace() ) );
101 
102  midi_params.append(0); // Initial MIDI paramater number
103  midi_delegates.append(_poti); // for the potentiometer
104  qDebug()<<"There are "<<midi_delegates.size()<<" Midi delegates";
105 
106  // Now construct the parameter setting menu
107  _cca = new JackMix::GUI::MidiControlChannelAssigner(QString("Set MIDI control parameter"),
108  "<qt>" + _in[0] + " &rarr; " + _out[0] + "</qt>",
109  QStringList() << "Gain",
110  midi_params,
111  this
112  );
113  connect( _cca, SIGNAL(assignParameters(QList<int>)), this, SLOT(update_midi_parameters(QList<int>)) );
114 }
