49 Knob::Knob( double v, double min, double max, int precision, double pagestep,
50  QWidget* p, QString valuestring, QColor indicator )
51  : AbstractSlider( v, min, max, precision, pagestep, p, valuestring )
52  , _timer( new QTimer( this ) )
53  , _show_value( false )

59  setFocusPolicy( Qt::TabFocus );

61  _timer->setInterval( 2000 ); // 2 seconds timeout for showing the value
62  _timer->setSingleShot( true );
63  connect( _timer, SIGNAL( timeout() ), this, SLOT( timeOut() ) );
64 }

91 void Knob::paintEvent( QPaintEvent* ) {
92  QPainter p( this );
93  p.setRenderHints( QPainter::Antialiasing );
94 
95  p.translate( width()/2.0, height()/2.0 );
96 
97  p.save();
98 
99  p.rotate( -240 );
100 
101  double radius = qMin( width(), height() )/2 - 4; // "Border" of 10 pixels
102 
103  // Draw Arc around whole area
104  //p.drawArc( QRectF( -radius, -radius, 2*radius, 2*radius ), 16* 0, 16* -300 );
105 
106  // Draw Knob
107  p.save();
108  p.setPen( Qt::NoPen );
109  {
110  QRadialGradient grad( QPointF( 0,0 ), radius, QPointF( 0, radius*0.7 ) );
111  grad.setColorAt( 0, _indicator );
112  grad.setColorAt( 1, _indicator.dark() );
113  grad.setSpread( QGradient::PadSpread );
114  p.setBrush( grad );
115  p.drawEllipse( QRectF( -radius*0.8, -radius*0.8, radius*1.6, radius*1.6 ) );
116  }
117  {
118  QRadialGradient grad( QPointF( 0,0 ), radius*0.60, QPointF( 0, radius*0.20 ) );
119  grad.setColorAt( 1, palette().color( QPalette::Highlight ) );
120  grad.setColorAt( 0, palette().color( QPalette::Highlight ).light() );
121  grad.setSpread( QGradient::PadSpread );
122  p.setBrush( grad );
123  p.drawEllipse( QRectF( -radius*0.65, -radius*0.65, radius*1.3, radius*1.3 ) );
124  }
125  p.restore();
126 
127  // Draw ticks
128  p.save();
129  p.setPen( palette().color( QPalette::ButtonText ) );
130  if ( _show_value ) {
131  for ( double a=_pagestep; a<dbmax; a+= _pagestep ) {
132  p.save();
133  p.rotate( dbtondb( a )*300 );
134  p.drawLine( QPointF( radius*0.80,0 ), QPointF( radius*0.85,0 ) );
135  p.restore();
136  }
137  for ( double a=-_pagestep; a>dbmin; a-= _pagestep ) {
138  p.save();
139  p.rotate( dbtondb( a )*300 );
140  p.drawLine( QPointF( radius*0.80,0 ), QPointF( radius*0.85,0 ) );
141  p.restore();
142  }
143  }
144  QList<double> _texts;
145  _texts << dbmin << dbmax;
146  if ( 0<dbmax && 0>dbmin )
147  _texts << 0.0;
148  QFont small = font();
149  small.setPointSizeF( qMax( qreal(5.0), font().pointSizeF()/2 ) );
150  p.setFont( small );
151  foreach( double a, _texts ) {
152  p.save();
153  p.rotate( dbtondb( a )*300 );
154  p.drawLine( QPointF( radius*0.80,0 ), QPointF( radius*0.92,0 ) );
155  p.translate( QPointF( radius*0.91,0 ) );
156  p.rotate( -dbtondb( a )*300 +240 );
157  QRectF rect(
158  0,0,
159  QFontMetrics( small ).width( _valuestring ),
160  QFontMetrics( small ).height() );
161  if ( dbtondb( a ) < 0.5 )
162  rect.translate( -QFontMetrics( small ).width( _valuestring ), 0 );
163  p.drawText( rect, Qt::AlignCenter, QString( _valuestring ).arg( a ) );
164  if ( a == 0.0 )
165  _nullclick = p.matrix().mapRect( rect ).toRect();
166  p.restore();
167  }
168  p.restore();
169 
170  // Draw highlight-line for the value
171  p.save();
172  p.rotate( 300 * dbtondb( _value ) );
173  QPen linepen( palette().color( QPalette::HighlightedText ) );
174  linepen.setWidthF( 3 );
175  linepen.setCapStyle( Qt::RoundCap );
176  p.setPen( linepen );
177  p.drawLine( QPointF( 0,0 ), QPointF( radius*0.71,0 ) );
178  p.restore();
179 
180  p.restore();
181 
182  if ( _show_value ) {
183 
184  QString tmp = QString::number( _value );
185  if ( tmp.contains( "." ) )
186  tmp = _valuestring.arg( tmp.left( tmp.indexOf( "." ) + _precision + 1 ) );
187  else
188  tmp = _valuestring.arg( tmp );
189 
190  QFontMetrics metrics( font() );
191  QRectF rect = metrics.boundingRect( tmp );
192  double x = rect.width() /2;
193  double y = qMax( int( radius ), height()/2 );
194  {
195  // Draw a 50% transparent rect behind the text...
196  p.save();
197  p.setPen( Qt::NoPen );
198  //p.setBrush( palette().color( QPalette::Window ) );
199  p.setOpacity( 0.75 );
200  p.setBrush( palette().color( QPalette::Base ) );
201  p.drawRoundRect( rect.translated( -x, y*0.9 ).adjusted( -2, 1, 2,-2 ), 20, 50 );
202  p.restore();
203  }
204  p.drawText( QPointF( -x, y*0.9 ), tmp );
205 
206  }

208 }

210 void Knob::mouseEvent( QMouseEvent* ev ) {
211  qDebug() << "Knob::mouseEvent(" << ev << ")";
212  qDebug() << " ev->x" << ev->x() << " ev->y" << ev->y();
213  QPointF pos( ev->pos() );
214  QPointF middle( width()/2.0, height()/2.0 );
215  QPointF rpos( pos-middle );
216  if ( abs( rpos ) > 10 ) {
217  qDebug() << " " << pos << " - " << middle << " = " << rpos;
218  qDebug() << " length is" << abs( rpos );
219  double alpha_sin = asin( rpos.y()/ abs( rpos ) )/M_PI*180;
220  qDebug() << " asin(rpos)" << alpha_sin;
221  double alpha_cos = acos( rpos.x()/ abs( rpos ) )/M_PI*180;
222  qDebug() << " acos(rpos)" << alpha_cos;
223  double alpha = alpha_cos * sgn( alpha_sin );
224  qDebug() << " alpha" << alpha;
225  alpha = degree( alpha + 240 );
226  if ( alpha > 330 )
227  alpha = 0;
228  alpha = qMin( alpha, 300.0 );
229  qDebug() << " alpha" << alpha;
230 
231  qDebug() << " value =" << alpha/300 << " ndbtodb =" << ndbtodb( alpha/300 );
232  value( ndbtodb( alpha/300 ) );
233  }
234 }
