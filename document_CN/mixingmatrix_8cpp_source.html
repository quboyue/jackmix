164  createControl( QStringList()<<*init, QStringList()<<*init );

167  }

209 void Widget::resizeEvent( QResizeEvent* ) {
210  setMinimumSize( sizeHint() );
211  qDebug( "MinimumSize = (%i,%i)", sizeHint().width(), sizeHint().height() );
212  int ins = _inchannels.size(), outs = _outchannels.size();
213  qDebug( "%i InChannels and %i OutChannels", ins, outs );

215 
216  int w=1, h=1;
217  if ( ins && outs ) {
218  w = width()/ins;

221 
222  if ( smallestElement().width()>w )
223  w = smallestElement().width();
224  if ( smallestElement().height()>h )

226 
227  if ( _direction == Horizontal )
228  //h=0;

231  _elements[ i ]->show();
232  }
233  else if ( _direction == Vertical )
234  //w=0;
235  for ( int i=0; i<_elements.size(); i++ ) {
236  _elements[ i ]->setGeometry( 0, _outchannels.indexOf( _elements[ i ]->out()[ 0 ] )*h, w, h*_elements[ i ]->outchannels() );
237  _elements[ i ]->show();
238  }
239  else
240  for ( int i=0; i<_elements.size(); i++ ) {
241  _elements[ i ]->setGeometry( _inchannels.indexOf( _elements[ i ]->in()[ 0 ] )*w, _outchannels.indexOf( _elements[ i ]->out()[ 0 ] )*h, w*_elements[ i ]->inchannels(), h*_elements[ i ]->outchannels() );
242  _elements[ i ]->show();
243  }

317 void Widget::renamechannels(QString old_name, QString new_name) {
318 
319  if (_direction == Vertical || _direction == Horizontal) {
320  // Input and output sliders will have same string in and out
321  Element *tmp = getResponsible(old_name, old_name);
322  if (tmp) {
323  tmp->renamechannels(old_name, new_name);
324  //tmp->repaint();
325  }
326 
327  } else {
328  if (_outchannels.indexOf(old_name) >= 0) { // it's an output we're changing
329  for (QStringList::Iterator it = _inchannels.begin();
330  it != _inchannels.end();
331  it++) {
332  qDebug() << "Matrix mixer: renaming (" << *it << ", " << old_name << ")";
333  Element *tmp = getResponsible( *it, old_name );
334  if ( tmp )
335  tmp->renamechannels(old_name, new_name);
336  }
337  } else { // an input name is changing: iterate over all outputs
338  for (QStringList::Iterator it = _outchannels.begin();
339  it != _outchannels.end();
340  it++) {
341  qDebug() << "Matrix mixer: renaming (" << old_name << ", " << *it << ")";
342  Element *tmp = getResponsible( old_name, *it );
343  if ( tmp )
344  tmp->renamechannels(old_name, new_name);
345  }
346  }
347  }
348 
349  // Now maintain the private channel lists
350 
351  int pos;
352  if ( (pos = _inchannels.indexOf(old_name)) >= 0 ) _inchannels[pos] = new_name;
353  if ( (pos = _outchannels.indexOf(old_name)) >= 0 ) _outchannels[pos] = new_name;
